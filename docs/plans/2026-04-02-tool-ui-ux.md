# Tool UI/UX Implementation Plan

> **Goal**: Achieve parity with Claude Code's compact, progressive-disclosure tool rendering.
> **Reference**: `docs/TOOL_UI_UX_CURRENTSTATE.md` (gap analysis)

---

## Overview

Transform smolbot's verbose per-tool rendering (64+ lines for 8 reads) into Claude Code's
collapsed group approach (~2 lines for 8 reads) with progressive disclosure.

### Current State
- Every tool gets a full bordered block (6-20 lines each)
- No grouping, no aggregation, no footer tracking
- 8 file reads consume ~1.6 screens

### Target State
- Consecutive collapsible tools grouped into natural-language summaries
- "Read 3 files, searched 2 patterns" instead of 3+2 separate blocks
- Ctrl+O to expand, footer shows running tool counts
- Individual tools show metadata (line counts, durations) not raw output

```go
package chat

import "strings"

// ToolClass determines rendering strategy for a tool call.
type ToolClass int

const (
    // ToolClassCollapsible tools can be grouped into a single summary line.
    // These are "read-like" operations that don't need individual attention.
    ToolClassCollapsible ToolClass = iota

    // ToolClassStandalone tools must render individually with full detail.
    // These are "write-like" operations where output matters.
    ToolClassStandalone
)

func (c ToolClass) String() string {
    switch c {
    case ToolClassCollapsible:
        return "collapsible"
    case ToolClassStandalone:
        return "standalone"
    default:
        return "unknown"
    }
}

// ToolKind identifies the semantic type of a tool for counting and display.
type ToolKind int

const (
    ToolKindRead ToolKind = iota   // file read operations
    ToolKindSearch                  // grep, web search
    ToolKindList                    // directory listing
    ToolKindBash                    // shell execution
    ToolKindWrite                   // file write
    ToolKindEdit                    // file edit (with diff)
    ToolKindMessage                 // channel messaging
    ToolKindFetch                   // web fetch
    ToolKindSpawn                   // agent spawn
    ToolKindOther                   // unknown/unclassified
)

func (k ToolKind) String() string {
    switch k {
    case ToolKindRead:
        return "read"
    case ToolKindSearch:
        return "search"
    case ToolKindList:
        return "list"
    case ToolKindBash:
        return "bash"
    case ToolKindWrite:
        return "write"
    case ToolKindEdit:
        return "edit"
    case ToolKindMessage:
        return "message"
    case ToolKindFetch:
        return "fetch"
    case ToolKindSpawn:
        return "spawn"
    case ToolKindOther:
        return "other"
    default:
        return "unknown"
    }
}

// ClassifyTool determines how a tool should be rendered and counted.
// Returns (class, kind) where class determines grouping strategy.
func ClassifyTool(name string) (ToolClass, ToolKind) {
    // Normalize: lowercase, strip prefixes/suffixes
    normalized := strings.ToLower(strings.TrimSpace(name))

    switch normalized {
    // Collapsible: read-like operations
    case "read_file", "readfile", "file_read":
        return ToolClassCollapsible, ToolKindRead
    case "list_dir", "listdir", "directory_list", "ls":
        return ToolClassCollapsible, ToolKindList
    case "web_search", "search", "grep":
        return ToolClassCollapsible, ToolKindSearch

    // Standalone: write-like operations (output matters)
    case "write_file", "writefile", "file_write":
        return ToolClassStandalone, ToolKindWrite
    case "edit_file", "editfile", "file_edit":
        return ToolClassStandalone, ToolKindEdit
    case "exec", "bash", "shell", "run":
        return ToolClassStandalone, ToolKindBash
    case "web_fetch", "fetch", "curl":
        return ToolClassStandalone, ToolKindFetch
    case "message":
        return ToolClassStandalone, ToolKindMessage
    case "spawn", "spawn_agent":
        return ToolClassStandalone, ToolKindSpawn

    default:
        // Unknown tools are standalone by default (safe: shows all detail)
        return ToolClassStandalone, ToolKindOther
    }
}

// IsCollapsible is a convenience method.
func IsCollapsible(name string) bool {
    class, _ := ClassifyTool(name)
    return class == ToolClassCollapsible
}
```

```go
package chat

import (
    "encoding/json"
    "strings"
    "time"
)

// ToolGroup accumulates consecutive collapsible tools for grouped rendering.
// This is the Go equivalent of Claude Code's GroupAccumulator.
type ToolGroup struct {
    // Raw counts (incremented on each Add)
    ReadCount    int
    SearchCount  int
    ListCount    int

    // Deduplicated tracking
    FilePaths     map[string]struct{}  // Unique file paths (for dedup display)
    SearchQueries []string             // Search patterns (for hint display)
    ListDirs      []string             // Listed directories

    // All tools in this group (for expanded rendering)
    Tools []ToolCall

    // Status flags
    IsActive bool  // At least one tool still running
    HasError bool  // At least one tool errored

    // Anti-jitter: max counts ever seen (only increases during streaming)
    // Display uses these, not raw counts, to prevent flickering.
    MaxReadCount   int
    MaxSearchCount int
    MaxListCount   int

    // Timing for hint display
    StartTime     time.Time  // When first tool started
    LastToolStart time.Time  // When most recent tool started (for hint)
}

// NewToolGroup creates an empty group ready for accumulation.
func NewToolGroup() *ToolGroup {
    return &ToolGroup{
        FilePaths:     make(map[string]struct{}),
        SearchQueries: make([]string, 0, 4),
        ListDirs:      make([]string, 0, 4),
        Tools:         make([]ToolCall, 0, 8),
        StartTime:     time.Now(),
    }
}

// Add incorporates a tool into the group.
// This is the main accumulation method called by CollapseTools.
func (g *ToolGroup) Add(tc ToolCall, kind ToolKind) {
    g.Tools = append(g.Tools, tc)

    // Update status flags
    switch strings.ToLower(tc.Status) {
    case "running":
        g.IsActive = true
        g.LastToolStart = time.Now()
    case "error":
        g.HasError = true
    }

    // Extract and deduplicate file paths
    if path := extractFilePath(tc.Input); path != "" {
        g.FilePaths[path] = struct{}{}
    }

    // Increment counters and update max values
    switch kind {
    case ToolKindRead:
        g.ReadCount++
        // Use deduplicated count for display (same file read 3x = "read 1 file")
        g.MaxReadCount = max(g.MaxReadCount, len(g.FilePaths))

    case ToolKindSearch:
        g.SearchCount++
        g.MaxSearchCount = max(g.MaxSearchCount, g.SearchCount)
        // Track query for hint line
        if query := extractSearchQuery(tc.Input); query != "" {
            g.SearchQueries = append(g.SearchQueries, query)
        }

    case ToolKindList:
        g.ListCount++
        g.MaxListCount = max(g.MaxListCount, g.ListCount)
        // Track directory for hint line
        if dir := extractFilePath(tc.Input); dir != "" {
            g.ListDirs = append(g.ListDirs, dir)
        }
    }
}

// Empty returns true if no tools have been added.
func (g *ToolGroup) Empty() bool {
    return len(g.Tools) == 0
}

// DisplayReadCount returns the count to show in UI (uses max for anti-jitter).
func (g *ToolGroup) DisplayReadCount() int {
    if g.MaxReadCount > 0 {
        return g.MaxReadCount
    }
    return len(g.FilePaths)
}

// DisplaySearchCount returns the count to show in UI.
func (g *ToolGroup) DisplaySearchCount() int {
    return g.MaxSearchCount
}

// DisplayListCount returns the count to show in UI.
func (g *ToolGroup) DisplayListCount() int {
    return g.MaxListCount
}

// CurrentOperation returns the most recent active tool for hint display.
// Returns nil if no tools are currently running.
func (g *ToolGroup) CurrentOperation() *ToolCall {
    for i := len(g.Tools) - 1; i >= 0; i-- {
        if strings.ToLower(g.Tools[i].Status) == "running" {
            return &g.Tools[i]
        }
    }
    return nil
}

// RunningCount returns how many tools are still running.
func (g *ToolGroup) RunningCount() int {
    count := 0
    for _, tc := range g.Tools {
        if strings.ToLower(tc.Status) == "running" {
            count++
        }
    }
    return count
}

// ErrorCount returns how many tools errored.
func (g *ToolGroup) ErrorCount() int {
    count := 0
    for _, tc := range g.Tools {
        if strings.ToLower(tc.Status) == "error" {
            count++
        }
    }
    return count
}

// --- JSON field extraction helpers ---
// These are also used by message.go but we define them here for self-containment.

func extractFilePath(input string) string {
    var data map[string]interface{}
    if err := json.Unmarshal([]byte(input), &data); err != nil {
        return ""
    }
    if path, ok := data["path"].(string); ok {
        return strings.TrimSpace(path)
    }
    return ""
}

func extractSearchQuery(input string) string {
    var data map[string]interface{}
    if err := json.Unmarshal([]byte(input), &data); err != nil {
        return ""
    }
    if query, ok := data["query"].(string); ok {
        return strings.TrimSpace(query)
    }
    // Also check "pattern" for grep-like tools
    if pattern, ok := data["pattern"].(string); ok {
        return strings.TrimSpace(pattern)
    }
    return ""
}
```

### Data Structures

```go
// CollapsedBlock represents either a collapsed group or a standalone tool.
// This is the output of the collapse algorithm.
type CollapsedBlock struct {
    IsGroup bool        // true = render as collapsed group, false = render individually
    Group   *ToolGroup  // Populated if IsGroup
    Tool    *ToolCall   // Populated if !IsGroup
    Kind    ToolKind    // Tool kind (for standalone) or dominant kind (for group)
}
```

### Algorithm: `CollapseTools`

The algorithm mirrors Claude Code's `collapseReadSearchGroups()`:

```go
// CollapseTools transforms a flat list of tools into grouped/standalone blocks.
//
// Rules:
// 1. Consecutive collapsible tools (read_file, list_dir, web_search) are grouped
// 2. Standalone tools (write_file, edit_file, exec, etc.) break the group
// 3. Each standalone tool becomes its own block
// 4. Groups with only 1 tool are still rendered as groups (consistent UX)
//
// Example:
//   Input:  [read, read, write, read, exec, read]
//   Output: [Group(2 reads), Standalone(write), Group(1 read), Standalone(exec), Group(1 read)]
func CollapseTools(tools []ToolCall) []CollapsedBlock {
    if len(tools) == 0 {
        return nil
    }

    blocks := make([]CollapsedBlock, 0, len(tools)/2+1)
    var currentGroup *ToolGroup

    // flushGroup emits the current group (if any) and resets
    flushGroup := func() {
        if currentGroup != nil && !currentGroup.Empty() {
            blocks = append(blocks, CollapsedBlock{
                IsGroup: true,
                Group:   currentGroup,
                Kind:    dominantKind(currentGroup),
            })
            currentGroup = nil
        }
    }

    for i := range tools {
        tc := &tools[i]  // Pointer to avoid copying
        class, kind := ClassifyTool(tc.Name)

        if class == ToolClassCollapsible {
            // Collapsible: add to current group (or start new one)
            if currentGroup == nil {
                currentGroup = NewToolGroup()
            }
            currentGroup.Add(*tc, kind)
        } else {
            // Standalone: flush any pending group, then emit standalone block
            flushGroup()
            blocks = append(blocks, CollapsedBlock{
                IsGroup: false,
                Tool:    tc,
                Kind:    kind,
            })
        }
    }

    // Flush any remaining group
    flushGroup()

    return blocks
}

// dominantKind returns the most common ToolKind in a group (for labeling).
func dominantKind(g *ToolGroup) ToolKind {
    maxCount := 0
    dominant := ToolKindRead

    if g.DisplayReadCount() > maxCount {
        maxCount = g.DisplayReadCount()
        dominant = ToolKindRead
    }
    if g.DisplaySearchCount() > maxCount {
        maxCount = g.DisplaySearchCount()
        dominant = ToolKindSearch
    }
    if g.DisplayListCount() > maxCount {
        dominant = ToolKindList
    }

    return dominant
}
```

```go
import (
    "fmt"
    "strings"

// Summary returns a past-tense natural-language description of completed work.
// Example: "Read 3 files, searched 2 patterns, listed 1 directory"
func (g *ToolGroup) Summary() string {
    return g.buildSummary(false)
}

// ActiveSummary returns a present-tense description for active operations.
// Example: "Reading 3 files, searching 2 patterns..."
func (g *ToolGroup) ActiveSummary() string {
    return g.buildSummary(true) + "..."
}

// buildSummary constructs the natural-language string.
// Uses max-counts for anti-jitter during streaming.
func (g *ToolGroup) buildSummary(active bool) string {
    var parts []string

    readCount := g.DisplayReadCount()
    searchCount := g.DisplaySearchCount()
    listCount := g.DisplayListCount()

    // Read operations
    if readCount > 0 {
        if active {
            parts = append(parts, pluralize("reading", "file", readCount))
        } else {
            parts = append(parts, pluralize("read", "file", readCount))
        }
    }

    // Search operations
    if searchCount > 0 {
        if active {
            parts = append(parts, pluralizeSuffix("searching", "pattern", searchCount))
        } else {
            parts = append(parts, pluralizeSuffix("searched", "pattern", searchCount))
        }
    }

    // List operations
    if listCount > 0 {
        if active {
            parts = append(parts, pluralizeDir("listing", listCount))
        } else {
            parts = append(parts, pluralizeDir("listed", listCount))
        }
    }

    if len(parts) == 0 {
        if active {
            return "working"
        }
        return "completed"
    }

    return strings.Join(parts, ", ")
}

// pluralize handles "verb N noun(s)" format.
// Examples: "read 1 file", "read 3 files"
func pluralize(verb, noun string, count int) string {
    if count == 1 {
        return fmt.Sprintf("%s %d %s", verb, count, noun)
    }
    return fmt.Sprintf("%s %d %ss", verb, count, noun)
}

// pluralizeSuffix handles verbs that need "for" prefix.
// Examples: "searched for 1 pattern", "searched for 3 patterns"
func pluralizeSuffix(verb, noun string, count int) string {
    if count == 1 {
        return fmt.Sprintf("%s for %d %s", verb, count, noun)
    }
    return fmt.Sprintf("%s for %d %ss", verb, count, noun)
}

// pluralizeDir handles directory pluralization (directory/directories).
// Examples: "listed 1 directory", "listed 3 directories"
func pluralizeDir(verb string, count int) string {
    if count == 1 {
        return fmt.Sprintf("%s %d directory", verb, count)
    }
    return fmt.Sprintf("%s %d directories", verb, count)
}
```

```go
// RenderToolGroup renders a collapsed group as a single compact line.
// This replaces 6-20 lines per tool with a single summary line.
//
// Output formats:
//   Active:  "● Reading 3 files, searching 2 patterns..."
//   Done:    "✓ Read 3 files, searched 2 patterns"
//   Error:   "✗ Read 3 files (1 failed)"
//
// When verbose=true (Ctrl+O pressed), returns empty string to signal
// that caller should render individual tools instead.
func RenderToolGroup(group *ToolGroup, t *theme.Theme, width int, verbose bool) string {
    if verbose {
        return ""  // Signal to render individual tools
    }

    if t == nil {
        return renderPlainToolGroup(group)
    }

    // Build status indicator (2 chars for alignment)
    indicator, indicatorColor := groupIndicator(group, t)
    indicatorText := lipgloss.NewStyle().
        Foreground(indicatorColor).
        Render(indicator)

    // Build summary text with proper capitalization
    var summary string
    if group.IsActive {
        summary = group.ActiveSummary()
    } else {
        summary = group.Summary()
    }
    // Capitalize first letter
    if len(summary) > 0 {
        summary = strings.ToUpper(summary[:1]) + summary[1:]
    }

    // Add error count suffix if any errors
    if group.HasError && group.ErrorCount() > 0 {
        errorSuffix := fmt.Sprintf(" (%d failed)", group.ErrorCount())
        summary += lipgloss.NewStyle().
            Foreground(t.Error).
            Render(errorSuffix)
    }

    // Main line: indicator + summary
    line := indicatorText + " " + summary

    // Add expansion hint (muted) if space permits
    hint := lipgloss.NewStyle().
        Foreground(t.TextMuted).
        Render("  (Ctrl+O to expand)")
    hintWidth := lipgloss.Width(hint)

    if width > 0 && lipgloss.Width(line)+hintWidth < width-4 {
        line += hint
    }

    return line
}

// groupIndicator returns the status character and color.
func groupIndicator(group *ToolGroup, t *theme.Theme) (string, color.Color) {
    if group.HasError {
        return "✗", t.Error
    }
    if group.IsActive {
        return "●", t.Warning  // Could animate with ◐◓◑◒ via spinnerFrame
    }
    return "✓", t.Success
}

// renderPlainToolGroup handles the case when theme is nil.
func renderPlainToolGroup(group *ToolGroup) string {
    var prefix string
    if group.HasError {
        prefix = "[!] "
    } else if group.IsActive {
        prefix = "[.] "
    } else {
        prefix = "[✓] "
    }

    var summary string
    if group.IsActive {
        summary = group.ActiveSummary()
    } else {
        summary = group.Summary()
    }

    return prefix + summary
}
```

```go
// RenderToolGroupWithSpinner renders with animated spinner for active groups.
// spinnerFrame cycles 0-3 to animate ◐→◓→◑→◒
func RenderToolGroupWithSpinner(group *ToolGroup, t *theme.Theme, width int, verbose bool, spinnerFrame int) string {
    if verbose || !group.IsActive {
        return RenderToolGroup(group, t, width, verbose)
    }

    if t == nil {
        return renderPlainToolGroup(group)
    }

    // Animated spinner
    frames := []string{"◐", "◓", "◑", "◒"}
    spinner := frames[spinnerFrame%len(frames)]
    indicatorText := lipgloss.NewStyle().
        Foreground(t.Warning).
        Render(spinner)

    summary := group.ActiveSummary()
    if len(summary) > 0 {
        summary = strings.ToUpper(summary[:1]) + summary[1:]
    }

    return indicatorText + " " + summary
}
```

```go
// renderToolBlocks renders tools using the collapse engine.
// In verbose mode, renders each tool individually.
// In collapsed mode, groups consecutive collapsible tools.
func (m *MessagesModel) renderToolBlocks(t *theme.Theme) []string {
    if len(m.tools) == 0 {
        return nil
    }

    if m.verbose {
        // Verbose mode: existing per-tool rendering
        result := make([]string, 0, len(m.tools))
        for i, tool := range m.tools {
            expanded := m.expandedTools[strconv.Itoa(i)]
            result = append(result, renderToolCall(tool, m.width, expanded))
        }
        return result
    }

    // Collapsed mode: use collapse engine
    collapsed := CollapseTools(m.tools)
    result := make([]string, 0, len(collapsed))

    for _, block := range collapsed {
        if block.IsGroup {
            // Render collapsed group
            groupLine := RenderToolGroupWithSpinner(block.Group, t, m.width, false, m.spinnerFrame)
            if groupLine != "" {
                result = append(result, groupLine)
            }
        } else {
            // Standalone tool: use enhanced compact rendering
            result = append(result, m.renderStandaloneToolCompact(block.Tool, t))
        }
    }

    return result
}
```

```go
// renderWriteCompact: "✓ Wrote config.json (142 lines)"
func renderWriteCompact(tc ToolCall, width int, t *theme.Theme) string {
    path := extractFilePath(tc.Input)
    baseName := filepath.Base(path)
    if baseName == "" || baseName == "." {
        baseName = path
    }

    // Count lines in content
    content := extractJSONField(tc.Input, "content")
    lineCount := strings.Count(content, "\n") + 1
    if content == "" {
        lineCount = 0
    }

    // Build status prefix
    indicator, color := toolIndicator(tc.Status, t)
    prefix := lipgloss.NewStyle().Foreground(color).Render(indicator)

    // Verb based on status
    var verb string
    switch strings.ToLower(tc.Status) {
    case "running":
        verb = "Writing"
```

```go
func (m *FooterModel) SetToolCounts(running, done, errored int) {

    m.toolsDone = done
    m.toolsError = errored
    m.syncToolActivity()
}

// IncrementToolRunning increments running count.
func (m *FooterModel) IncrementToolRunning() {
    m.toolsRunning++
    m.syncToolActivity()
}

// ToolDone moves one from running to done.
func (m *FooterModel) ToolDone() {
    if m.toolsRunning > 0 {
        m.toolsRunning--
    }
    m.toolsDone++
    m.syncToolActivity()
}

// ToolError moves one from running to error.
func (m *FooterModel) ToolError() {
    if m.toolsRunning > 0 {
        m.toolsRunning--
    }
    m.toolsError++
    m.syncToolActivity()
}

// ResetToolCounts clears all tool counters.
func (m *FooterModel) ResetToolCounts() {
    m.toolsRunning = 0
    m.toolsDone = 0
    m.toolsError = 0
    m.toolActivityText = ""
    m.toolActivityWidth = 0
}

// syncToolActivity builds the tool activity display string.
func (m *FooterModel) syncToolActivity() {
    t := theme.Current()
    if t == nil {
        m.toolActivityText = ""
        m.toolActivityWidth = 0
        return
    }

    var parts []string

    if m.toolsDone > 0 {
        done := lipgloss.NewStyle().
            Foreground(t.Success).
            Render(fmt.Sprintf("✓%d", m.toolsDone))
        parts = append(parts, done)
    }

    if m.toolsRunning > 0 {
        running := lipgloss.NewStyle().
            Foreground(t.Warning).
            Render(fmt.Sprintf("●%d", m.toolsRunning))
        parts = append(parts, running)
    }

    if m.toolsError > 0 {
        errored := lipgloss.NewStyle().
            Foreground(t.Error).
            Render(fmt.Sprintf("✗%d", m.toolsError))
        parts = append(parts, errored)
    }

    if len(parts) == 0 {
        m.toolActivityText = ""
        m.toolActivityWidth = 0
        return
    }

    m.toolActivityText = strings.Join(parts, " ")
    m.toolActivityWidth = lipgloss.Width(m.toolActivityText)
}
```

### Anti-Flicker

Claude Code uses `MIN_HINT_DISPLAY_MS = 700ms` to prevent flickering. Implement:

```go
// In ToolGroup struct
type ToolGroup struct {
    // ... existing fields ...

    // Hint anti-flicker: remember the last displayed hint
    lastHintTool    string    // Tool ID of last hint
    lastHintTime    time.Time // When hint was first shown
}

const minHintDisplayMs = 700 * time.Millisecond

// CurrentOperationForHint returns the tool to show in hint line.
// Implements anti-flicker: shows same hint for at least 700ms.
func (g *ToolGroup) CurrentOperationForHint() *ToolCall {
    current := g.CurrentOperation()
    if current == nil {
        return nil
    }

    now := time.Now()

    // If same tool, keep showing
    if g.lastHintTool == current.ID {
        return current
    }

    // If different tool, check if minimum time elapsed
    if now.Sub(g.lastHintTime) < minHintDisplayMs && g.lastHintTool != "" {
        // Find the old tool and keep showing it
        for i := range g.Tools {
            if g.Tools[i].ID == g.lastHintTool {
                return &g.Tools[i]
            }
        }
    }

    // Update to new hint
    g.lastHintTool = current.ID
    g.lastHintTime = now
    return current
}
```
